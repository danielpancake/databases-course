-- 1. Find the universities who are located in Russia
SELECT name FROM University WHERE location='Russia';

-- 2. Find the countries of universities who are not located in Russia but teaches Russian natives
SELECT location FROM University INNER JOIN Student ON University.name = Student.university_name WHERE location<>'Russia' AND native_lang='ru';

-- 3. Find student ids who studies in Innopolis University
SELECT id FROM Student WHERE university_name='Innopolis University';

-- 4. Find pair of course names and universities names if course credit is more than 5
SELECT name,university_name FROM Course WHERE credits>5;

-- 5. Find the universities who teaches English native speakers
SELECT DISTINCT University.name FROM University INNER JOIN Student ON University.name = Student.university_name WHERE native_lang='en';
